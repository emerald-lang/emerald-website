Emerald

A language agnostic templating engine

sample.emr

html
  head
    title Emerald Example

    styles
      "css/main.css"
      "css/footer.css"

    scripts
      "js/main.js"

  body
    header
      h1 Emerald Language
      h3 The people

    main
      each people as person
      section
        h2 |person.name|
        p  |person.info|

    footer
      p Copyright 2016, Emerald Language

context.json

{
  people: [
    {name: "Andrew", info: "Plays the sax."},
    {name: "Dave", info: "Draws cool stuff."}
    {name: "Yu Chen Hou"}
  ]
}
